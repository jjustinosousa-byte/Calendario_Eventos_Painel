df_salvar.drop(columns=["N.º"]) salvar_dados(df_final) st.success("Alterações salvas com sucesso!") st.rerun() st.divider() st.write("### 🗑️ Excluir Evento") col_select, col_botao = st.columns([3, 1]) with col_select: lista_eventos = [""] + df_exibicao["NOME EVENTO"].tolist() evento_selecionado = st.selectbox("Selecione o evento para exclusão:", lista_eventos, index=None, 2: CADASTRO DE EVENTO --- elif pagina == "Cadastro de evento": st.title("📝 Cadastro de Novo Evento") if 'sucesso_cadastro' not in st.session_state: st.session_state['sucesso_cadastro'] = False
